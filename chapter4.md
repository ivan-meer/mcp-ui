```python
    loc_counts = {ext: 0 for ext in extensions} # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º —Å—á–µ—Ç—á–∏–∫–∏ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è
                                                # –ù–∞–ø—Ä–∏–º–µ—Ä, {'.py': 0, '.js': 0}
    loc_counts['total_processed_files'] = 0 # –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω–æ –ø–æ—Å—á–∏—Ç–∞–µ–º, —Å–∫–æ–ª—å–∫–æ —Ñ–∞–π–ª–æ–≤ –æ–±—Ä–∞–±–æ—Ç–∞–ª–∏

    for dirpath, dirnames, filenames in os.walk(folder_path):
        for file_name in filenames: # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –≤—Å–µ —Ñ–∞–π–ª—ã –≤ —Ç–µ–∫—É—â–µ–π –ø–∞–ø–∫–µ
            for ext in extensions: # –ü—Ä–æ–≤–µ—Ä—è–µ–º –∫–∞–∂–¥–æ–µ –∏–Ω—Ç–µ—Ä–µ—Å—É—é—â–µ–µ –Ω–∞—Å —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–µ
                if file_name.endswith(ext):
                    # –£—Ä–∞, —Ñ–∞–π–ª —Å –Ω—É–∂–Ω—ã–º —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–µ–º –Ω–∞–π–¥–µ–Ω!
                    loc_counts['total_processed_files'] += 1
                    
                    file_path = os.path.join(dirpath, file_name)
                    # os.path.join() –ø—Ä–∞–≤–∏–ª—å–Ω–æ —Å–æ–µ–¥–∏–Ω—è–µ—Ç –ø—É—Ç—å –∫ –ø–∞–ø–∫–µ –∏ –∏–º—è —Ñ–∞–π–ª–∞,
                    # —á—Ç–æ–±—ã –ø–æ–ª—É—á–∏—Ç—å –ø–æ–ª–Ω—ã–π –ø—É—Ç—å –∫ —Ñ–∞–π–ª—É. –≠—Ç–æ –≤–∞–∂–Ω–æ –¥–ª—è
                    # –∫—Ä–æ—Å—Å-–ø–ª–∞—Ç—Ñ–æ—Ä–º–µ–Ω–Ω–æ—Å—Ç–∏ (—Ä–∞–±–æ—Ç–∞–µ—Ç –∏ –Ω–∞ Windows —Å '\', –∏ –Ω–∞ Linux/macOS —Å '/').

                    try:
                        with open(file_path, 'r', encoding='utf-8') as f:
                            # 'with open(...) as f:' - —ç—Ç–æ –º–µ–Ω–µ–¥–∂–µ—Ä –∫–æ–Ω—Ç–µ–∫—Å—Ç–∞.
                            # –û–Ω –≥–∞—Ä–∞–Ω—Ç–∏—Ä—É–µ—Ç, —á—Ç–æ —Ñ–∞–π–ª –±—É–¥–µ—Ç –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ –∑–∞–∫—Ä—ã—Ç
                            # –ø–æ—Å–ª–µ —Ç–æ–≥–æ, –∫–∞–∫ –º—ã —Å –Ω–∏–º –ø–æ—Ä–∞–±–æ—Ç–∞–µ–º (–¥–∞–∂–µ –µ—Å–ª–∏ –ø—Ä–æ–∏–∑–æ–π–¥–µ—Ç –æ—à–∏–±–∫–∞).
                            # 'r' - —Ä–µ–∂–∏–º —á—Ç–µ–Ω–∏—è —Ñ–∞–π–ª–∞.
                            # encoding='utf-8' - –æ—á–µ–Ω—å –≤–∞–∂–Ω—ã–π –ø–∞—Ä–∞–º–µ—Ç—Ä! –û–Ω —É–∫–∞–∑—ã–≤–∞–µ—Ç,
                            # –≤ –∫–∞–∫–æ–π –∫–æ–¥–∏—Ä–æ–≤–∫–µ –Ω—É–∂–Ω–æ —á–∏—Ç–∞—Ç—å —Ñ–∞–π–ª. UTF-8 - —Å–∞–º–∞—è
                            # —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–Ω–∞—è –∫–æ–¥–∏—Ä–æ–≤–∫–∞, –ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞—é—â–∞—è –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Å–∏–º–≤–æ–ª–æ–≤
                            # —Ä–∞–∑–Ω—ã—Ö —è–∑—ã–∫–æ–≤. –ë–µ–∑ —è–≤–Ω–æ–≥–æ —É–∫–∞–∑–∞–Ω–∏—è –∫–æ–¥–∏—Ä–æ–≤–∫–∏ –º–æ–≥—É—Ç –±—ã—Ç—å –æ—à–∏–±–∫–∏
                            # –ø—Ä–∏ —á—Ç–µ–Ω–∏–∏ —Ñ–∞–π–ª–æ–≤, —Å–æ–¥–µ—Ä–∂–∞—â–∏—Ö –Ω–µ-–∞–Ω–≥–ª–∏–π—Å–∫–∏–µ —Å–∏–º–≤–æ–ª—ã.

                            lines = f.readlines() # –ß–∏—Ç–∞–µ–º –≤—Å–µ —Å—Ç—Ä–æ–∫–∏ –∏–∑ —Ñ–∞–π–ª–∞ –≤ —Å–ø–∏—Å–æ–∫
                            num_lines = len(lines) # –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—Ä–æ–∫ - —ç—Ç–æ –¥–ª–∏–Ω–∞ —Å–ø–∏—Å–∫–∞
                            loc_counts[ext] += num_lines # –î–æ–±–∞–≤–ª—è–µ–º –∫ —Å—á–µ—Ç—á–∏–∫—É –¥–ª—è –¥–∞–Ω–Ω–æ–≥–æ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è
                    
                    except UnicodeDecodeError:
                        # –≠—Ç–∞ –æ—à–∏–±–∫–∞ –≤–æ–∑–Ω–∏–∫–∞–µ—Ç, –µ—Å–ª–∏ —Ñ–∞–π–ª —Å–æ—Ö—Ä–∞–Ω–µ–Ω –Ω–µ –≤ UTF-8
                        # –∏ —Å–æ–¥–µ—Ä–∂–∏—Ç —Å–∏–º–≤–æ–ª—ã, –∫–æ—Ç–æ—Ä—ã–µ –Ω–µ –º–æ–≥—É—Ç –±—ã—Ç—å –ø—Ä–æ—á–∏—Ç–∞–Ω—ã —Å —ç—Ç–æ–π –∫–æ–¥–∏—Ä–æ–≤–∫–æ–π.
                        print(f"–ù–µ —É–¥–∞–ª–æ—Å—å –ø—Ä–æ—á–∏—Ç–∞—Ç—å —Ñ–∞–π–ª (–æ—à–∏–±–∫–∞ –∫–æ–¥–∏—Ä–æ–≤–∫–∏): {file_path}")
                        # –ú–æ–∂–Ω–æ –ø–æ–ø—Ä–æ–±–æ–≤–∞—Ç—å –¥—Ä—É–≥–∏–µ –∫–æ–¥–∏—Ä–æ–≤–∫–∏ –∏–ª–∏ –ø—Ä–æ—Å—Ç–æ –ø—Ä–æ–ø—É—Å—Ç–∏—Ç—å —Ñ–∞–π–ª
                        try:
                            with open(file_path, 'r', encoding='latin-1') as f:
                                lines = f.readlines()
                                num_lines = len(lines)
                                loc_counts[ext] += num_lines
                                print(f"–§–∞–π–ª {file_path} —É—Å–ø–µ—à–Ω–æ –ø—Ä–æ—á–∏—Ç–∞–Ω —Å –∫–æ–¥–∏—Ä–æ–≤–∫–æ–π latin-1.")
                        except Exception as e:
                            print(f"–ù–µ —É–¥–∞–ª–æ—Å—å –ø—Ä–æ—á–∏—Ç–∞—Ç—å —Ñ–∞–π–ª {file_path} –∏ —Å –¥—Ä—É–≥–∏–º–∏ –∫–æ–¥–∏—Ä–æ–≤–∫–∞–º–∏: {e}")
                    except Exception as e:
                        # –õ–æ–≤–∏–º –¥—Ä—É–≥–∏–µ –≤–æ–∑–º–æ–∂–Ω—ã–µ –æ—à–∏–±–∫–∏ –ø—Ä–∏ —Ä–∞–±–æ—Ç–µ —Å —Ñ–∞–π–ª–æ–º (–Ω–∞–ø—Ä–∏–º–µ—Ä, –Ω–µ—Ç –ø—Ä–∞–≤ –¥–æ—Å—Ç—É–ø–∞)
                        print(f"–û—à–∏–±–∫–∞ –ø—Ä–∏ –æ–±—Ä–∞–±–æ—Ç–∫–µ —Ñ–∞–π–ª–∞ {file_path}: {e}")
                    
                    break # –§–∞–π–ª –æ–±—Ä–∞–±–æ—Ç–∞–Ω –¥–ª—è –æ–¥–Ω–æ–≥–æ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è, –≤—ã—Ö–æ–¥–∏–º –∏–∑ –≤–Ω—É—Ç—Ä–µ–Ω–Ω–µ–≥–æ —Ü–∏–∫–ª–∞ –ø–æ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è–º
                          # (–µ—Å–ª–∏ —Ñ–∞–π–ª .py, –Ω–µ –Ω—É–∂–Ω–æ –ø—Ä–æ–≤–µ—Ä—è—Ç—å, .js –ª–∏ –æ–Ω)
    
    return loc_counts # –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Å–ª–æ–≤–∞—Ä—å —Å –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ–º —Å—Ç—Ä–æ–∫ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ —Ç–∏–ø–∞ —Ñ–∞–π–ª–∞
```

**–†–∞–∑–±–∏—Ä–∞–µ–º –∫–ª—é—á–µ–≤—ã–µ –º–æ–º–µ–Ω—Ç—ã:**

*   `loc_counts = {ext: 0 for ext in extensions}`: –ú—ã —Å–æ–∑–¥–∞–µ–º —Å–ª–æ–≤–∞—Ä—å, –≥–¥–µ –∫–ª—é—á–∞–º–∏ –±—É–¥—É—Ç —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è (–Ω–∞–ø—Ä–∏–º–µ—Ä, `'.py'`), –∞ –∑–Ω–∞—á–µ–Ω–∏—è–º–∏ ‚Äì –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—Ä–æ–∫ –∫–æ–¥–∞ –¥–ª—è —ç—Ç–∏—Ö —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–π. –ò—Å–ø–æ–ª—å–∑—É–µ–º "—Å–ª–æ–≤–∞—Ä–Ω–æ–µ –≤–∫–ª—é—á–µ–Ω–∏–µ" (dictionary comprehension) –¥–ª—è –∫—Ä–∞—Ç–∫–æ—Å—Ç–∏.
*   `file_name.endswith(ext)`: –≠—Ç–æ—Ç –º–µ—Ç–æ–¥ —Å—Ç—Ä–æ–∫–∏ –ø—Ä–æ–≤–µ—Ä—è–µ—Ç, –∑–∞–∫–∞–Ω—á–∏–≤–∞–µ—Ç—Å—è –ª–∏ –∏–º—è —Ñ–∞–π–ª–∞ `file_name` –Ω–∞ —É–∫–∞–∑–∞–Ω–Ω–æ–µ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–µ `ext`.
*   `file_path = os.path.join(dirpath, file_name)`: –û—á–µ–Ω—å –≤–∞–∂–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è! –û–Ω–∞ "—É–º–Ω–æ" —Å–æ–µ–¥–∏–Ω—è–µ—Ç –ø—É—Ç—å –∫ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏ (`dirpath`) –∏ –∏–º—è —Ñ–∞–π–ª–∞ (`file_name`), –∏—Å–ø–æ–ª—å–∑—É—è –ø—Ä–∞–≤–∏–ª—å–Ω—ã–π —Ä–∞–∑–¥–µ–ª–∏—Ç–µ–ª—å –¥–ª—è —Ç–µ–∫—É—â–µ–π –æ–ø–µ—Ä–∞—Ü–∏–æ–Ω–Ω–æ–π —Å–∏—Å—Ç–µ–º—ã (`/` –¥–ª—è Linux/macOS, `\` –¥–ª—è Windows). –í—Å–µ–≥–¥–∞ –∏—Å–ø–æ–ª—å–∑—É–π `os.path.join()` –¥–ª—è —Ä–∞–±–æ—Ç—ã —Å –ø—É—Ç—è–º–∏, —á—Ç–æ–±—ã —Ç–≤–æ–π –∫–æ–¥ –±—ã–ª –ø–µ—Ä–µ–Ω–æ—Å–∏–º—ã–º.
*   `with open(file_path, 'r', encoding='utf-8') as f:`:
    *   `with ... as ...`: –≠—Ç–æ –∫–æ–Ω—Å—Ç—Ä—É–∫—Ü–∏—è –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è "–º–µ–Ω–µ–¥–∂–µ—Ä –∫–æ–Ω—Ç–µ–∫—Å—Ç–∞". –û–Ω–∞ –≥–∞—Ä–∞–Ω—Ç–∏—Ä—É–µ—Ç, —á—Ç–æ —Ñ–∞–π–ª –±—É–¥–µ—Ç –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ –∑–∞–∫—Ä—ã—Ç –ø–æ—Å–ª–µ —Ç–æ–≥–æ, –∫–∞–∫ –±–ª–æ–∫ –∫–æ–¥–∞ –≤–Ω—É—Ç—Ä–∏ `with` –∑–∞–≤–µ—Ä—à–∏—Ç—Å—è, –¥–∞–∂–µ –µ—Å–ª–∏ –ø—Ä–∏ —á—Ç–µ–Ω–∏–∏ —Ñ–∞–π–ª–∞ –ø—Ä–æ–∏–∑–æ–π–¥–µ—Ç –æ—à–∏–±–∫–∞. –≠—Ç–æ –æ—á–µ–Ω—å —É–¥–æ–±–Ω–æ –∏ –±–µ–∑–æ–ø–∞—Å–Ω–æ.
    *   `'r'`: –†–µ–∂–∏–º –æ—Ç–∫—Ä—ã—Ç–∏—è —Ñ–∞–π–ª–∞ "read" (—á—Ç–µ–Ω–∏–µ).
    *   `encoding='utf-8'`: –£–∫–∞–∑—ã–≤–∞–µ—Ç –∫–æ–¥–∏—Ä–æ–≤–∫—É —Ñ–∞–π–ª–∞. –°–æ–∑–¥–∞–π —Ç–∞–∫—É—é —Å—Ç—Ä—É–∫—Ç—É—Ä—É –∏ —Ñ–∞–π–ª—ã —Å —É–∫–∞–∑–∞–Ω–Ω—ã–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ–º —Å—Ç—Ä–æ–∫ (–ø—Ä–æ—Å—Ç–æ –Ω–∞–∂–º–∏ Enter –Ω–µ—Å–∫–æ–ª—å–∫–æ —Ä–∞–∑ –≤ —Ñ–∞–π–ª–∞—Ö).
