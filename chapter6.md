## üìú –§–∏–Ω–∞–ª—å–Ω—ã–π –∫–æ–¥ –ø—Ä–∏–ª–æ–∂–µ–Ω–∏—è

–í–æ—Ç –ø–æ–ª–Ω—ã–π –∫–æ–¥ –Ω–∞—à–µ–≥–æ "–ê–Ω–∞–ª–∏–∑–∞—Ç–æ—Ä–∞ –ö–æ–¥–∞". –≠—Ç–æ—Ç —Å–∫—Ä–∏–ø—Ç –æ–±—ä–µ–¥–∏–Ω—è–µ—Ç –≤—Å–µ, —á—Ç–æ –º—ã —Å–æ–∑–¥–∞–ª–∏ –≤ –ø—Ä–µ–¥—ã–¥—É—â–∏—Ö –≥–ª–∞–≤–∞—Ö: –≥—Ä–∞—Ñ–∏—á–µ—Å–∫–∏–π –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å –Ω–∞ Tkinter, —Ñ—É–Ω–∫—Ü–∏–∏ –¥–ª—è –≤—ã–±–æ—Ä–∞ –ø–∞–ø–∫–∏ –∏ —Ñ—É–Ω–∫—Ü–∏–∏ –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞ —Ñ–∞–π–ª–æ–≤.

```python
import tkinter as tk
from tkinter import filedialog, messagebox
import os

# --- –ì–ª–æ–±–∞–ª—å–Ω–∞—è –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –ø—É—Ç–∏ –∫ –≤—ã–±—Ä–∞–Ω–Ω–æ–π –ø–∞–ø–∫–µ ---
selected_folder_path = ""

# --- –§—É–Ω–∫—Ü–∏–∏ –∞–Ω–∞–ª–∏–∑–∞ (–∏–∑ –ì–ª–∞–≤—ã 4) ---
def count_total_files(folder_path):
    """
    –†–µ–∫—É—Ä—Å–∏–≤–Ω–æ –ø–æ–¥—Å—á–∏—Ç—ã–≤–∞–µ—Ç –æ–±—â–µ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –≤—Å–µ—Ö —Ñ–∞–π–ª–æ–≤ –≤ —É–∫–∞–∑–∞–Ω–Ω–æ–π –ø–∞–ø–∫–µ
    –∏ –µ–µ –ø–æ–¥–ø–∞–ø–∫–∞—Ö.
    """
    total_files = 0
    # os.walk –≥–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç –∫–æ—Ä—Ç–µ–∂–∏ (–ø—É—Ç—å_–∫_–ø–∞–ø–∫–µ, —Å–ø–∏—Å–æ–∫_–∏–º–µ–Ω_–ø–æ–¥–ø–∞–ø–æ–∫, —Å–ø–∏—Å–æ–∫_–∏–º–µ–Ω_—Ñ–∞–π–ª–æ–≤)
    for _, _, filenames in os.walk(folder_path):
        total_files += len(filenames) # –£–≤–µ–ª–∏—á–∏–≤–∞–µ–º —Å—á–µ—Ç—á–∏–∫ –Ω–∞ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ñ–∞–π–ª–æ–≤ –≤ —Ç–µ–∫—É—â–µ–π –ø–∞–ø–∫–µ
    return total_files

def count_lines_of_code(folder_path, extensions=['.py', '.js']):
    """
    –ü–æ–¥—Å—á–∏—Ç—ã–≤–∞–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—Ä–æ–∫ –∫–æ–¥–∞ –≤ —Ñ–∞–π–ª–∞—Ö —Å —É–∫–∞–∑–∞–Ω–Ω—ã–º–∏ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è–º–∏
    –≤ –¥–∞–Ω–Ω–æ–π –ø–∞–ø–∫–µ –∏ –µ–µ –ø–æ–¥–ø–∞–ø–∫–∞—Ö.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ª–æ–≤–∞—Ä—å, –≥–¥–µ –∫–ª—é—á–∏ - —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è, –∞ –∑–Ω–∞—á–µ–Ω–∏—è - –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—Ä–æ–∫.
    –¢–∞–∫–∂–µ –≤–∫–ª—é—á–∞–µ—Ç 'total_processed_files' - –æ–±—â–µ–µ —á–∏—Å–ª–æ –æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã—Ö —Ñ–∞–π–ª–æ–≤.
    """
    loc_counts = {ext: 0 for ext in extensions} # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è —Å–ª–æ–≤–∞—Ä—è –¥–ª—è —Å—á–µ—Ç—á–∏–∫–æ–≤
    loc_counts['total_processed_files'] = 0    # –°—á–µ—Ç—á–∏–∫ –æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã—Ö —Ñ–∞–π–ª–æ–≤

    for dirpath, _, filenames in os.walk(folder_path): # –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –æ–±—Ö–æ–¥ –ø–∞–ø–æ–∫
        for file_name in filenames: # –ü–µ—Ä–µ–±–æ—Ä –≤—Å–µ—Ö —Ñ–∞–π–ª–æ–≤ –≤ —Ç–µ–∫—É—â–µ–π –ø–∞–ø–∫–µ
            for ext in extensions: # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ –∫–∞–∂–¥–æ–º—É –∏–∑ –∑–∞–¥–∞–Ω–Ω—ã—Ö —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–π
                if file_name.endswith(ext):
                    loc_counts['total_processed_files'] += 1
                    file_path = os.path.join(dirpath, file_name) # –§–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–∏–µ –ø–æ–ª–Ω–æ–≥–æ –ø—É—Ç–∏ –∫ —Ñ–∞–π–ª—É
                    
                    try:
                        # –ü—ã—Ç–∞–µ–º—Å—è –æ—Ç–∫—Ä—ã—Ç—å –∏ –ø—Ä–æ—á–∏—Ç–∞—Ç—å —Ñ–∞–π–ª —Å –∫–æ–¥–∏—Ä–æ–≤–∫–æ–π UTF-8
                        with open(file_path, 'r', encoding='utf-8') as f:
                            lines = f.readlines() # –ß–∏—Ç–∞–µ–º –≤—Å–µ —Å—Ç—Ä–æ–∫–∏
                            loc_counts[ext] += len(lines) # –î–æ–±–∞–≤–ª—è–µ–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—Ä–æ–∫ –∫ —Å—á–µ—Ç—á–∏–∫—É
                    except UnicodeDecodeError:
                        # –ï—Å–ª–∏ UTF-8 –Ω–µ —Å—Ä–∞–±–æ—Ç–∞–ª–∞, –ø—Ä–æ–±—É–µ–º latin-1 (—á–∞—Å—Ç–æ –ø–æ–º–æ–≥–∞–µ—Ç)
                        try:
                            with open(file_path, 'r', encoding='latin-1') as f:
                                lines = f.readlines()
                                loc_counts[ext] += len(lines)
                                print(f"–§–∞–π–ª {file_path} —É—Å–ø–µ—à–Ω–æ –ø—Ä–æ—á–∏—Ç–∞–Ω —Å –∫–æ–¥–∏—Ä–æ–≤–∫–æ–π latin-1.")
                        except Exception as e_inner:
                             # –ï—Å–ª–∏ –∏ latin-1 –Ω–µ –ø–æ–º–æ–≥–ª–∞, –≤—ã–≤–æ–¥–∏–º —Å–æ–æ–±—â–µ–Ω–∏–µ
                             print(f"–ù–µ —É–¥–∞–ª–æ—Å—å –ø—Ä–æ—á–∏—Ç–∞—Ç—å —Ñ–∞–π–ª {file_path} (–ø—Ä–æ–±–æ–≤–∞–ª–∏ UTF-8, Latin-1): {e_inner}")
                    except Exception as e_outer:
                        # –õ–æ–≤–∏–º –¥—Ä—É–≥–∏–µ –≤–æ–∑–º–æ–∂–Ω—ã–µ –æ—à–∏–±–∫–∏ –ø—Ä–∏ —Ä–∞–±–æ—Ç–µ —Å —Ñ–∞–π–ª–æ–º
                        print(f"–û—à–∏–±–∫–∞ –ø—Ä–∏ –æ–±—Ä–∞–±–æ—Ç–∫–µ —Ñ–∞–π–ª–∞ {file_path}: {e_outer}")
                    
                    break # –§–∞–π–ª –æ–±—Ä–∞–±–æ—Ç–∞–Ω –¥–ª—è –æ–¥–Ω–æ–≥–æ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è, –ø–µ—Ä–µ—Ö–æ–¥–∏–º –∫ —Å–ª–µ–¥—É—é—â–µ–º—É —Ñ–∞–π–ª—É
    return loc_counts

# --- –§—É–Ω–∫—Ü–∏–∏-–æ–±—Ä–∞–±–æ—Ç—á–∏–∫–∏ –¥–ª—è –∫–Ω–æ–ø–æ–∫ GUI (–∏–∑ –ì–ª–∞–≤ 3 –∏ 5) ---
def selected_folder_action():
    """
    –í—ã–∑—ã–≤–∞–µ—Ç—Å—è –ø—Ä–∏ –Ω–∞–∂–∞—Ç–∏–∏ –∫–Ω–æ–ø–∫–∏ "–í—ã–±—Ä–∞—Ç—å –ø–∞–ø–∫—É".
    –û—Ç–∫—Ä—ã–≤–∞–µ—Ç –¥–∏–∞–ª–æ–≥ –≤—ã–±–æ—Ä–∞ –ø–∞–ø–∫–∏ –∏ –æ–±–Ω–æ–≤–ª—è–µ—Ç –º–µ—Ç–∫—É —Å –ø—É—Ç–µ–º.
    –¢–∞–∫–∂–µ —Å–±—Ä–∞—Å—ã–≤–∞–µ—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –ø—Ä–µ–¥—ã–¥—É—â–µ–≥–æ –∞–Ω–∞–ª–∏–∑–∞.
    """
    global selected_folder_path # –†–∞–±–æ—Ç–∞–µ–º —Å –≥–ª–æ–±–∞–ª—å–Ω–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π
    folder_path_temp = filedialog.askdirectory() # –û—Ç–∫—Ä—ã–≤–∞–µ–º –¥–∏–∞–ª–æ–≥

    if folder_path_temp: # –ï—Å–ª–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –≤—ã–±—Ä–∞–ª –ø–∞–ø–∫—É
        selected_folder_path = folder_path_temp
        lbl_folder_path.config(text=selected_folder_path) # –û–±–Ω–æ–≤–ª—è–µ–º –º–µ—Ç–∫—É —Å –ø—É—Ç–µ–º
        # –°–±—Ä–∞—Å—ã–≤–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –ø—Ä–µ–¥—ã–¥—É—â–µ–≥–æ –∞–Ω–∞–ª–∏–∑–∞, —Ç–∞–∫ –∫–∞–∫ –≤—ã–±—Ä–∞–Ω–∞ –Ω–æ–≤–∞—è –ø–∞–ø–∫–∞
        lbl_total_files.config(text="–í—Å–µ–≥–æ —Ñ–∞–π–ª–æ–≤: -")
        lbl_py_loc.config(text="–°—Ç—Ä–æ–∫ –∫–æ–¥–∞ Python: -")
        lbl_js_loc.config(text="–°—Ç—Ä–æ–∫ –∫–æ–¥–∞ JavaScript: -")
    else: # –ï—Å–ª–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –æ—Ç–º–µ–Ω–∏–ª –≤—ã–±–æ—Ä
        if not selected_folder_path: # –ï—Å–ª–∏ –∏ –¥–æ —ç—Ç–æ–≥–æ –Ω–∏—á–µ–≥–æ –Ω–µ –±—ã–ª–æ –≤—ã–±—Ä–∞–Ω–æ
            lbl_folder_path.config(text="–ü–∞–ø–∫–∞ —Ç–∞–∫ –∏ –Ω–µ –±—ã–ª–∞ –≤—ã–±—Ä–∞–Ω–∞.")
        # –ï—Å–ª–∏ –ø–∞–ø–∫–∞ –±—ã–ª–∞ –≤—ã–±—Ä–∞–Ω–∞ —Ä–∞–Ω–µ–µ, –∞ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –Ω–∞–∂–∞–ª "–û—Ç–º–µ–Ω–∞",
        # —Ç–æ selected_folder_path –∏ —Ç–µ–∫—Å—Ç –º–µ—Ç–∫–∏ –æ—Å—Ç–∞—é—Ç—Å—è –ø—Ä–µ–∂–Ω–∏–º–∏.

def analyze_repository_action():
    """
    –í—ã–∑—ã–≤–∞–µ—Ç—Å—è –ø—Ä–∏ –Ω–∞–∂–∞—Ç–∏–∏ –∫–Ω–æ–ø–∫–∏ "–ê–Ω–∞–ª–∏–∑–∏—Ä–æ–≤–∞—Ç—å!".
    –ü—Ä–æ–≤–µ—Ä—è–µ—Ç, –≤—ã–±—Ä–∞–Ω–∞ –ª–∏ –ø–∞–ø–∫–∞, –∑–∞—Ç–µ–º –∑–∞–ø—É—Å–∫–∞–µ—Ç —Ñ—É–Ω–∫—Ü–∏–∏ –∞–Ω–∞–ª–∏–∑–∞
    –∏ –æ—Ç–æ–±—Ä–∞–∂–∞–µ—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –≤ GUI.
    """
    if not selected_folder_path: # –ü—Ä–æ–≤–µ—Ä–∫–∞, –≤—ã–±—Ä–∞–Ω–∞ –ª–∏ –ø–∞–ø–∫–∞
        messagebox.showerror("–û—à–∏–±–∫–∞", "–ü–æ–∂–∞–ª—É–π—Å—Ç–∞, —Å–Ω–∞—á–∞–ª–∞ –≤—ã–±–µ—Ä–∏—Ç–µ –ø–∞–ø–∫—É –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞!")
        # –£–±–µ–¥–∏–º—Å—è, —á—Ç–æ –ø–æ–ª—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ —Å–±—Ä–æ—à–µ–Ω—ã, –µ—Å–ª–∏ –ø–∞–ø–∫–∞ –Ω–µ –≤—ã–±—Ä–∞–Ω–∞
        lbl_total_files.config(text="–í—Å–µ–≥–æ —Ñ–∞–π–ª–æ–≤: -")
        lbl_py_loc.config(text="–°—Ç—Ä–æ–∫ –∫–æ–¥–∞ Python: -")
        lbl_js_loc.config(text="–°—Ç—Ä–æ–∫ –∫–æ–¥–∞ JavaScript: -")
        return # –í—ã—Ö–æ–¥ –∏–∑ —Ñ—É–Ω–∫—Ü–∏–∏

    # –ü—Ä–æ–≤–µ—Ä–∫–∞, —Å—É—â–µ—Å—Ç–≤—É–µ—Ç –ª–∏ –ø—É—Ç—å –∏ —è–≤–ª—è–µ—Ç—Å—è –ª–∏ –æ–Ω –ø–∞–ø–∫–æ–π
    if not os.path.exists(selected_folder_path) or not os.path.isdir(selected_folder_path):
        messagebox.showerror("–û—à–∏–±–∫–∞", f"–í—ã–±—Ä–∞–Ω–Ω—ã–π –ø—É—Ç—å –Ω–µ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç –∏–ª–∏ –Ω–µ —è–≤–ª—è–µ—Ç—Å—è –ø–∞–ø–∫–æ–π:\n{selected_folder_path}")
        return
        
    try:
        # –ó–∞–ø—É—Å–∫–∞–µ–º –∞–Ω–∞–ª–∏–∑
        total_files_count = count_total_files(selected_folder_path)
        loc_results = count_lines_of_code(selected_folder_path, extensions=['.py', '.js'])

        # –û—Ç–æ–±—Ä–∞–∂–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã
        lbl_total_files.config(text=f"–í—Å–µ–≥–æ —Ñ–∞–π–ª–æ–≤: {total_files_count}")
        
        # –ë–µ–∑–æ–ø–∞—Å–Ω–æ –ø–æ–ª—É—á–∞–µ–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—Ä–æ–∫, –∏—Å–ø–æ–ª—å–∑—É—è .get() —Å–æ –∑–Ω–∞—á–µ–Ω–∏–µ–º –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é 0
        py_lines = loc_results.get('.py', 0)
        js_lines = loc_results.get('.js', 0)
        
        lbl_py_loc.config(text=f"–°—Ç—Ä–æ–∫ –∫–æ–¥–∞ Python: {py_lines}")
        lbl_js_loc.config(text=f"–°—Ç—Ä–æ–∫ –∫–æ–¥–∞ JavaScript: {js_lines}")

        messagebox.showinfo("–ê–Ω–∞–ª–∏–∑ –∑–∞–≤–µ—Ä—à–µ–Ω", f"–ê–Ω–∞–ª–∏–∑ –ø–∞–ø–∫–∏ '{selected_folder_path}' —É—Å–ø–µ—à–Ω–æ –∑–∞–≤–µ—Ä—à–µ–Ω!")

    except Exception as e:
        # –û–±—Ä–∞–±–æ—Ç–∫–∞ –Ω–µ–ø—Ä–µ–¥–≤–∏–¥–µ–Ω–Ω—ã—Ö –æ—à–∏–±–æ–∫ –≤–æ –≤—Ä–µ–º—è –∞–Ω–∞–ª–∏–∑–∞
        messagebox.showerror("–û—à–∏–±–∫–∞ –∞–Ω–∞–ª–∏–∑–∞", f"–ü—Ä–æ–∏–∑–æ—à–ª–∞ –æ—à–∏–±–∫–∞ –≤–æ –≤—Ä–µ–º—è –∞–Ω–∞–ª–∏–∑–∞:\n{e}")
        lbl_total_files.config(text="–í—Å–µ–≥–æ —Ñ–∞–π–ª–æ–≤: –û—à–∏–±–∫–∞")
        lbl_py_loc.config(text="–°—Ç—Ä–æ–∫ –∫–æ–¥–∞ Python: –û—à–∏–±–∫–∞")
        lbl_js_loc.config(text="–°—Ç—Ä–æ–∫ –∫–æ–¥–∞ JavaScript: –û—à–∏–±–∫–∞")

# --- –°–æ–∑–¥–∞–Ω–∏–µ –≥–ª–∞–≤–Ω–æ–≥–æ –æ–∫–Ω–∞ –∏ –≤–∏–¥–∂–µ—Ç–æ–≤ GUI (–∏–∑ –ì–ª–∞–≤ 2, 3, 5) ---
root = tk.Tk()
root.title("–ê–Ω–∞–ª–∏–∑–∞—Ç–æ—Ä –ö–æ–¥–∞ v1.0") # –ú–æ–∂–Ω–æ –∏ –≤–µ—Ä—Å–∏—é —É–∫–∞–∑–∞—Ç—å :)
root.geometry("600x450") # –®–∏—Ä–∏–Ω–∞ x –í—ã—Å–æ—Ç–∞

# –§—Ä–µ–π–º –¥–ª—è –∫–Ω–æ–ø–æ–∫ –∏ –ø—É—Ç–∏ (–¥–ª—è –ª—É—á—à–µ–π –æ—Ä–≥–∞–Ω–∏–∑–∞—Ü–∏–∏)
frame_selection = tk.Frame(root)
frame_selection.pack(pady=10, padx=10, fill=tk.X)

btn_select_folder = tk.Button(frame_selection, text="1. –í—ã–±–µ—Ä–∏—Ç–µ –ø–∞–ø–∫—É –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞", command=selected_folder_action)
btn_select_folder.pack(fill=tk.X) # –†–∞—Å—Ç—è–Ω—É—Ç—å –ø–æ —à–∏—Ä–∏–Ω–µ —Ñ—Ä–µ–π–º–∞

lbl_folder_path = tk.Label(frame_selection, text="–ü–∞–ø–∫–∞ –Ω–µ –≤—ã–±—Ä–∞–Ω–∞", relief="sunken", wraplength=580, justify="left")
```
