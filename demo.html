Добро пожаловать в MCP UI SDK

MCP UI SDK позволяет создавать интерактивные пользовательские интерфейсы, которые встраиваются в Model Context Protocol серверы. Это открывает новые возможности для создания богатых, интерактивных инструментов для AI агентов.

Интеграция с MCP сервером

1. Установка зависимостей

npm install @mcp-ui/server @mcp-ui/client

2. Создание HTML ресурса на сервере

import { createHtmlResource } from '@mcp-ui/server';

// В вашем MCP tool
const resourceBlock = createHtmlResource({
  uri: 'ui://my-dashboard/1',
  content: { 
    type: 'rawHtml', 
    htmlString: '<h1>Hello World</h1>' 
  },
  delivery: 'text'
});

return {
  content: [resourceBlock]
};

3. Рендеринг на клиенте

import { HtmlResource } from '@mcp-ui/client';

function MyComponent({ resource }) {
  const handleUiAction = async (tool, params) => {
    // Обработка действий пользователя
    console.log('Tool called:', tool, params);
  };

  return (
    <HtmlResource 
      resource={resource}
      onUiAction={handleUiAction}
      renderMode="secure" // или "iframe"
    />
  );
}

4. Обработка событий

// В HTML ресурсе
<button data-tool="save_data" data-params='{"id": 123}'>
  Сохранить
</button>

// В режиме iframe используйте postMessage
window.parent.postMessage({
  tool: 'save_data',
  params: { id: 123 }
}, '*');
